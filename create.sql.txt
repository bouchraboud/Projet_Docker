CREATE TABLE ETUDIANTS (
    id SERIAL PRIMARY KEY,
    nom VARCHAR(255) NOT NULL,
    date_de_creation DATE NOT NULL DEFAULT CURRENT_DATE
);

CREATE TABLE NOTES (
    id SERIAL PRIMARY KEY,
    id_etudiant BIGINT NOT NULL,
    nom_du_cours VARCHAR(255) NOT NULL,
    valeur_de_note DOUBLE PRECISION NOT NULL,
    FOREIGN KEY (id_etudiant) REFERENCES ETUDIANTS(id) ON DELETE CASCADE
);
